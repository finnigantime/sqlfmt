--------------------------------------------------
SELECT
  CASE account_type
    WHEN 'free' THEN ''
    WHEN 'silver' THEN rank.silver
    WHEN 'platinum' THEN rank.platinum
    WHEN 'vip' THEN rank.vip
    ELSE rank.normal
    END AS account_rank
FROM
  accounts AS a
  JOIN rank AS r
    ON a.account_id = r.account_id
    AND a.account_type = r.account_type
