--------------------------------------------------
SELECT
  *
FROM
  dbs.accounts AS parent
  JOIN dbs.child
    ON parent.account_id = child.parent_account_id
    AND parent.account_id = ?
  JOIN (
    SELECT
      *
    FROM
      dbs.account_group
    WHERE
      is_deleted = false
  ) AS g
    ON child.group_id = g.group_id
ORDER BY group_name ASC
